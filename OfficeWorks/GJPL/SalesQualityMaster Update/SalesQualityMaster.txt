QUALITYDESCRIPTION = replace(QUALITYDESCRIPTION,'~','"'),
QUALITYSHORTDESCRIPTION = replace(QUALITYSHORTDESCRIPTION,'~','"')
where 1=1
and (QUALITYDESCRIPTION like '%~%' or QUALITYSHORTDESCRIPTION like '%~%')

'M'||DECODE(PARTYTYPE,'PARTY','P','BROKER','B','TRANSPORTER','T')||SUBSTR(PARTYNAME,1,1)||LPAD((rank() over (PARTITION by PARTYTYPE, SUBSTR(PARTYNAME,1,1) order by partyname)),3,'0') PARTYCODE, 
'M'||PARTYCODE PARTYCODE1, PARTYCODE OLDPARTYCODE, '01-APR-2020' EFFECTIVEDATE, UPPER(PARTYNAME) PARTTYNAME, UPPER(PARTYSHORTNAME) PARTYSHORTNAME,

PARTYEMAIL, DECODE(PARTYWEBSITE,'WEBSITE','',PARTYWEBSITE) PARTWEBSITE, UPPER(PARTYCONTACTPERSONNAME) PARTYCONTACTPERSONNAME , BANKNAME, BANKBRANCH, BANKACNO, BANKACTYPE,
